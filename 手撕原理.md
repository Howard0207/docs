# 手撕原理

## apply

```js
Function.prototype.myApply = function (context, arr) {
	const ctx = Object(context) || window;
	ctx.fn = this;
	let result;
	if (Array.isArray(arr)) {
		result = eval('ctx.fn(' + arr + ')');
	} else {
		result = ctx.fn();
	}
	delete ctx.fn;
	return result;
}
```

## call

```js
Function.prototype.mcall = function () {
	var args = Array.prototype.slice.call(arguments);
	var ctx = Object(args.shift()) || window;
	ctx.fn = this;
	var result = eval('ctx.fn(' + args + ')')
	delete ctx.fn;
	return result;
}
```

## bind

```js
Function.prototype.bind = function () {
	var args = Array.prototype.slice.call(arguments);
	var ctx = Object(args.shift()) || window;
	var result;
	ctx.fn = this;
	return function () {
		var nargs = Array.prototype.slice.call(arguments);
		var params = [].concat(args, nargs);
		result = eval('ctx.fn(' + params + ')')
		delete ctx.fn;
		return result;
	}
}
```

## deepClone

```js
function isObject(obj) {
	return (typeof obj === 'object' || typeof obj === 'function') && obj !== null;
}

function deepClone(obj, map = new WeakMap()) {
	if (!isObject(obj) || map.get(obj)) {
		return obj;
	}
	const constructor = object.constructor;
	if (/^(Regexp|Date)$/i.test(constructor.name)) {
		return new constructor(obj);
	}
	var result = Array.isArray(obj) ? [] : {};
	map.set(obj, true);
	for (let key in obj) {
		if (Object.prototype.hasOwnProperty.call(obj, key)) {
			result[key] = deepClone(obj[key], map);
		}
	}
	return result;
}
```

## Object.create

```js
Object.mycreate = function(proto, properties) {
    function F() {};
    F.prototype = proto;
    if(properties) {
        Object.defineProperties(F, properties);
    }
    return new F();
}
```
